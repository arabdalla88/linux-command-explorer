# commands-db.txt

:ls
يعرض محتوى المجلد الحالي.
أمثلة:
  ls              # عرض الملفات والمجلدات
  ls -l           # عرض بشكل مفصل
  ls -a           # عرض الملفات المخفية

:cd
يستخدم للتنقل بين المجلدات.
أمثلة:
  cd /home        # الدخول إلى مجلد home
  cd ..           # الرجوع إلى المجلد السابق
  cd              # العودة إلى المجلد الرئيسي (home)

:mkdir
يستخدم لإنشاء مجلد جديد.
أمثلة:
  mkdir test      # إنشاء مجلد باسم test
  mkdir -p a/b/c  # إنشاء عدة مجلدات متداخلة دفعة واحدة

:touch
يستخدم لإنشاء ملف فارغ.
أمثلة:
  touch file.txt  # إنشاء ملف نصي فارغ
  touch a b c     # إنشاء عدة ملفات دفعة واحدة

:rm
يستخدم لحذف الملفات أو المجلدات.
أمثلة:
  rm file.txt     # حذف ملف
  rm -r folder/   # حذف مجلد ومحتوياته
  rm -f file.txt  # حذف دون تأكيد

:cp
يستخدم لنسخ الملفات أو المجلدات.
أمثلة:
  cp file1 file2        # نسخ ملف
  cp -r dir1 dir2       # نسخ مجلد بمحتوياته

:mv
يستخدم لنقل أو إعادة تسمية الملفات.
أمثلة:
  mv old.txt new.txt    # إعادة تسمية ملف
  mv file.txt /tmp/     # نقل ملف إلى مجلد آخر

:cat
يعرض محتوى ملف نصي.
أمثلة:
  cat file.txt          # عرض محتوى الملف
  cat file1 file2       # دمج عرض ملفين معًا

:echo
يستخدم لطباعة نص في الطرفية.
أمثلة:
  echo "Hello World"    # طباعة نص عادي
  echo $HOME            # طباعة قيمة متغير

:pwd
يعرض المسار الكامل للمجلد الحالي.
أمثلة:
  pwd                   # طباعة المسار الكامل للمكان اللي أنت واقف فيه
